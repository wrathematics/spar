32  // sample a vector of length `len` with values coming from the interval
33  // [0, n-1]
34  template <typename INDEX>
35  static std::vector<INDEX> res_sampler(const uint32_t seed, const INDEX n,
36  const INDEX len, const bool zero_based_indices=true)
37  {
38  const INDEX offset = (INDEX) 1 - zero_based_indices;
39 
40  std::vector<INDEX> x(len);
41 
42  #pragma omp for simd
43  for (INDEX i=0; i<len; i++)
44  x[i] = i + offset;
45 
46  std::mt19937 mt(seed);
47 
48  for (INDEX i=len; i<n; i++)
49  {
50  std::uniform_int_distribution<INDEX> dist(0, i);
51  INDEX j = dist(mt);
52 
53  if (j < len)
54  x[j] = i + offset;
55  }
56 
57  std::sort(x.begin(), x.end());
58 
59  return x;
60  }

64  template <typename INDEX, typename SCALAR>
65  static inline void rand_fill_exact(const uint32_t seed, const INDEX nnz,
66  spmat<INDEX, SCALAR> &x)
67  {
68  const INDEX nrows = x.nrows();
69  const INDEX ncols = x.ncols();
70 
71  INDEX *I = x.index_ptr();
72  INDEX *P = x.col_ptr();
73  SCALAR *X = x.data_ptr();
74 
75  const INDEX plen = ncols + 1;
76 
77  std::vector<INDEX> indices = internals::res_sampler(seed, nrows*ncols, nnz);
78  INDEX col = 0;
79  for (INDEX ind=0; ind<nnz; ind++)
80  {
81  INDEX i = indices[ind] % nrows;
82  INDEX j = (indices[ind] - i) / nrows;
83 
84  I[ind] = i;
85  X[ind] = 1;
86 
87  if (col == j)
88  {
89  P[col] = ind;
90  col++;
91  }
92 
93  while (col < j)
94  {
95  P[col] = ind;
96  col++;
97  }
98  }
99 
100  for (INDEX ind=col; ind<plen; ind++)
101  P[ind] += nnz;
102 
103  x.update_nnz();
104  }

108  template <typename INDEX, typename SCALAR>
109  static inline void rand_fill_approx(const uint32_t seed,
110  const float prop_dense, const INDEX nnz, spmat<INDEX, SCALAR> &x)
111  {
112  const INDEX nrows = x.nrows();
113  const INDEX ncols = x.ncols();
114 
115  INDEX *I = x.index_ptr();
116  INDEX *P = x.col_ptr();
117  SCALAR *X = x.data_ptr();
118 
119  std::mt19937 mt(seed);
120  std::bernoulli_distribution dist(prop_dense);
121 
122  INDEX ind = 0;
123  for (INDEX j=0; j<ncols; j++)
124  {
125  for (INDEX i=0; i<nrows; i++)
126  {
127  auto p = dist(mt);
128  if (p)
129  {
130  I[ind] = i;
131  X[ind] = 1;
132  ind++;
133  }
134 
135  if (ind >= nnz)
136  break;
137  }
138 
139  P[j] = ind;
140  }
141 
142  P[ncols] = P[ncols-1];
143 
144  x.update_nnz();
145  }
146  }

168  template <typename INDEX, typename SCALAR>
169  static inline spmat<INDEX, SCALAR> rand(const uint32_t seed,
170  const float prop_dense, const INDEX nrows, const INDEX ncols,
171  const bool exact=true)
172  {
173  const INDEX nnz = (INDEX) (std::min(std::max(prop_dense, 0.f), 1.f) * nrows*ncols);
174  spmat<INDEX, SCALAR> x(nrows, ncols, nnz);
175  if (nnz == 0)
176  return x;
177 
178  if (exact)
179  internals::rand_fill_exact(seed, nnz, x);
180  else
181  internals::rand_fill_approx(seed, prop_dense, nnz, x);
182 
183  return x;
184  }

221  template <typename INDEX, typename SCALAR>
222  static inline spmat<INDEX, SCALAR> bandish(const uint32_t seed,
223  const INDEX nrows, const INDEX ncols)
224  {
225  const INDEX slen = (INDEX) nrows/2;
226  spvec<INDEX, SCALAR> s(slen);
227 
228  const INDEX xlen = (INDEX) slen * std::min(nrows, ncols);
229  spmat<INDEX, SCALAR> x(nrows, ncols, xlen);
230 
231  std::mt19937 mt(seed);
232  for (INDEX j=0; j<ncols; j++)
233  {
234  s.zero();
235 
236  for (INDEX i=0; i<nrows; i++)
237  {
238  const float p = internals::fudge(i, j, nrows);
239  std::binomial_distribution<int> dist(1, p);
240 
241  SCALAR d = dist(mt);
242  if (d > 0)
243  s.insert(i, 1);
244  }
245 
246  x.insert(j, s);
247  }
248 
249  return x;
250  }

271  template <typename INDEX>
272  static inline INDEX banded_nnz(const INDEX band, const INDEX nrows,
273  const INDEX ncols)
274  {
275  const INDEX m = nrows>ncols?nrows:ncols;
276  const INDEX n = nrows>ncols?ncols:nrows;
277 
278  INDEX nnz = n + ntri(n-1)-ntri(n-band) + ntri(m-1)-ntri(m-band) - (m-1-n);
279  if (m == n)
280  nnz -= (m-1 - n);
281 
282  return nnz;
283  }

300  template <typename INDEX, typename SCALAR>
301  static inline spmat<INDEX, SCALAR> banded(const INDEX band,
302  const INDEX nrows, const INDEX ncols)
303  {
304  const INDEX nnz = internals::banded_nnz(band, nrows, ncols);
305  spmat<INDEX, SCALAR> x(nrows, ncols, nnz);
306 
307  INDEX *I = x.index_ptr();
308  INDEX *P = x.col_ptr();
309  SCALAR *X = x.data_ptr();
310 
311  INDEX pos = 0;
312  INDEX col_pos = 1;
313  INDEX i = 0, j = 0;
314 
315  P[0] = 0;
316 
317  while (i < nrows)
318  {
319  if (i == j || (i < j && j < i + band) || (i > j && i < j + band))
320  {
321  I[pos] = i;
322  X[pos] = 1;
323  pos++;
324  i++;
325  }
326  else if (i >= band + j)
327  {
328  i = (i>band?i-band:0);
329  j++;
330 
331  P[col_pos] = pos;
332  col_pos++;
333 
334  if (j >= ncols)
335  break;
336  }
337  else
338  i++;
339  }
340 
341  x.update_nnz();
342  return x;
343  }
