60  static inline eigen_t<INDEX, SCALAR> spmat_to_eigen(const spmat<INDEX, SCALAR> &x)
61  {
62  const INDEX m = x.nrows();
63  const INDEX n = x.ncols();
64  const INDEX nnz = x.get_nnz();
65 
66  eigen_t<INDEX, SCALAR> s(m, n);
67  s.makeCompressed();
68  s.resizeNonZeros(nnz);
69 
70  arraytools::copy(nnz, x.data_ptr(), s.valuePtr());
71  arraytools::copy(nnz, x.index_ptr(), s.innerIndexPtr());
72  arraytools::copy(n+1, x.col_ptr(), s.outerIndexPtr());
73

135  static inline dvec<INDEX, SCALAR> eigen_to_dvec(const INDEX col, const eigen_t<INDEX, SCALAR> &s)
136  {
137  const INDEX m = s.rows();
138  const INDEX n = s.cols();
139  const INDEX nnz = s.nonZeros();
140 
141  const SCALAR *X = spmat_X(s);
142  const INDEX *I = spmat_I(s);
143  const INDEX *P = spmat_P(s);
144 
145  dvec<INDEX, SCALAR> d(m);
146 
147  const INDEX start = P[col];
148  const INDEX end = (col == n) ? nnz : P[col+1];
149  for (INDEX i=start; i<end; i++)
150  d[ I[i] ] = X[i];
151 
152  return d;

171  template <typename INDEX, typename SCALAR>
172  static inline void col(const INDEX j, const conv::eigen_t<INDEX, SCALAR> &x, spvec<INDEX, SCALAR> &s)
173  {
174  const INDEX *I = x.innerIndexPtr();
175  const INDEX *P = x.outerIndexPtr();
176  const SCALAR *X = x.valuePtr();
177 
178  const INDEX ind = P[j];
179  if (P[j + 1] == ind)
180  {
181  s.zero();
182  return;
183  }
184 
185  const INDEX col_nnz = P[j + 1] - ind;
186  s.set(col_nnz, I + ind, X + ind);
