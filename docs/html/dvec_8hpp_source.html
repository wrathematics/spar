281 void dvec<INDEX, SCALAR>::set(const INDEX_SRC nnz_, const INDEX_SRC *I_, const SCALAR_SRC *X_)
282 {
283  zero();
284 
285  INDEX_SRC top = I_[nnz_-1];
286  if (top > len)
287  resize(top + 1);
288 
289  for (INDEX_SRC i=0; i<nnz_; i++)

291 
292  nnz = nnz_;
293 }
