179 template <typename INDEX, typename SCALAR>
180 spar::spmat<INDEX, SCALAR>& spar::spmat<INDEX, SCALAR>::operator=(const spmat<INDEX, SCALAR>& x)
181 {
182  this->cleanup();
183 
184  m = x.nrows();
185  n = x.ncols();
186 
187  nnz = x.get_nnz();
188  len = x.get_len();
189  plen = n + 1;
190 
191  if (len == 0)
192  return *this;
193 
194  arraytools::alloc(len, &I);
195  arraytools::alloc(n+1, &P);
196  arraytools::alloc(len, &X);
197  arraytools::check_allocs(I, P, X);
198 
199  arraytools::copy(len, x.index_ptr(), I);
200  arraytools::copy(n+1, x.col_ptr(), P);
201  arraytools::copy(len, x.data_ptr(), X);
202 
203  return *this;
204 }

500 // convert "column pointer" P into column index
501 template <typename INDEX, typename SCALAR>
502 INDEX* spar::spmat<INDEX, SCALAR>::csc2coo()
503 {
504  INDEX j = 0;
505  INDEX ind = 0;
506 
507  INDEX* CI;
508  arraytools::alloc(len, &CI);
509  if (CI == NULL)
510  {
511  cleanup();
512  throw std::bad_alloc();
513  }
514 
515  for (INDEX c=0; c<plen; c++)
516  {
517  INDEX diff = P[c+1] - P[c];
518 
519  while (diff > 0)
520  {
521  CI[ind] = j;
522 
523  ind++;
524  diff--;
525  }
526 
527  j++;
528  }
529 
530  return CI;
531 }

535 template <typename INDEX, typename SCALAR>
536 void spar::spmat<INDEX, SCALAR>::insert_spvec(const INDEX col, const spar::spvec<INDEX, SCALAR> &x)
537 {
538  const INDEX *xI = x.index_ptr();
539  const SCALAR *xX = x.data_ptr();
540 
541  INDEX ind = P[col];
542  const INDEX xnnz = x.get_nnz();
543  for (INDEX xind=0; xind<xnnz; xind++)
544  {
545  I[ind] = xI[xind];
546  X[ind] = xX[xind];
547 
548  ind++;
549  }
550 
551  for (INDEX ind=col+1; ind<plen; ind++)
552  P[ind] += xnnz;
553 
554  nnz += xnnz;
555 }
