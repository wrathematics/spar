95  template <typename INDEX, typename SCALAR>
96  static inline void s4col_to_spvec(const int col_ind, SEXP s4, spvec<INDEX, SCALAR> &s)
97  {
98  SEXP s4_X = internal::sexp::get_x_from_s4(s4);
99  SEXP s4_I = internal::sexp::get_i_from_s4(s4); // len == nnz
100  SEXP s4_P = internal::sexp::get_p_from_s4(s4);
101 
102  const int start_ind = INTEGER(s4_P)[col_ind];
103  const int col_len = INTEGER(s4_P)[col_ind+1] - start_ind;
104 
105  s.set(col_len, INTEGER(s4_I) + start_ind, REAL(s4_X) + start_ind);
106  }

110  static inline spvec<INDEX, SCALAR> s4col_to_spvec(const int col_ind, SEXP s4)
111  {

114 
115  spvec<INDEX, SCALAR> s(col_len * spar::internal::defs::MEM_FUDGE_ELT_FAC);
116  s4col_to_spvec(col_ind, s4, s);

139  template <typename INDEX, typename SCALAR>
140  static inline SEXP spmat_to_s4(const spmat<INDEX, SCALAR> &s)
141  {
142  SEXP s4_class, s4;
143  SEXP s4_i, s4_p, s4_Dim, s4_Dimnames, s4_x, s4_factors;
144 
145  const INDEX m = s.nrows();
146  const INDEX n = s.ncols();
147  const INDEX nnz = s.get_nnz();
148  const INDEX p_len = n+1;
149 
150  PROTECT(s4_i = allocVector(INTSXP, nnz));
151  arraytools::copy(nnz, s.index_ptr(), INTEGER(s4_i));
152 
153  PROTECT(s4_p = allocVector(INTSXP, p_len));
154  arraytools::copy(p_len, s.col_ptr(), INTEGER(s4_p));
155 
156  PROTECT(s4_Dim = allocVector(INTSXP, 2));
157  INTEGER(s4_Dim)[0] = (int) m;
158  INTEGER(s4_Dim)[1] = (int) n;
159 
160  PROTECT(s4_Dimnames = allocVector(VECSXP, 2));
161  SET_VECTOR_ELT(s4_Dimnames, 0, R_NilValue);
162  SET_VECTOR_ELT(s4_Dimnames, 1, R_NilValue);
163 
164  PROTECT(s4_x = allocVector(REALSXP, nnz));
165  arraytools::copy(nnz, s.data_ptr(), REAL(s4_x));
166 
167  PROTECT(s4_factors = allocVector(VECSXP, 0));
168 
169  PROTECT(s4_class = MAKE_CLASS("dgCMatrix"));
170  PROTECT(s4 = NEW_OBJECT(s4_class));
171  SET_SLOT(s4, install("i"), s4_i);
172  SET_SLOT(s4, install("p"), s4_p);
173  SET_SLOT(s4, install("Dim"), s4_Dim);
174  SET_SLOT(s4, install("Dimnames"), s4_Dimnames);
175  SET_SLOT(s4, install("x"), s4_x);
176  SET_SLOT(s4, install("factors"), s4_factors);
177 
178  UNPROTECT(8);
179  return s4;
180  }
181  }

205  template <typename INDEX, typename SCALAR>
206  static inline INDEX max_col_nnz(const SEXP x)
207  {
208  INDEX m, n;
209  spar::internal::sexp::get_dim_from_s4(x, &m, &n);
210 
211  SEXP P = spar::internal::sexp::get_p_from_s4(x);
212 
213  INDEX max_nnz = 0;
214  for (INDEX col=0; col<n+1; col++)
215  {
216  INDEX col_nnz = spar::internal::sexp::get_col_len_from_s4(col, P);
217  if (col_nnz > max_nnz)
218  max_nnz = col_nnz;
219  }
220 
221  return max_nnz;
222  }
