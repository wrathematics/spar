145  template <class SPMAT, typename INDEX, typename SCALAR>
146  static inline spmat<INDEX, SCALAR> gather(const int root, const SPMAT &x, MPI_Comm comm=MPI_COMM_WORLD)
147  {
148  mpi::err::check_size(comm);
149  const bool receiving = (root == mpi::REDUCE_TO_ALL || root == mpi::get_rank(comm));
150 
151  INDEX m, n;
152  internal::get::dim<INDEX, SCALAR>(x, &m, &n);
153 
154  // setup
155  const INDEX len = internal::get::max_col_nnz<INDEX, SCALAR>(x) * spar::internal::defs::MEM_FUDGE_ELT_FAC;
156  spvec<INDEX, SCALAR> a(len);
157  spmat<INDEX, SCALAR> s(m, n, 0);
158 
159 
160  int size = mpi::get_size(comm);
161  dvec<int, int> counts(size);
162  dvec<int, int> displs(size);
163  displs[0] = 0;
164 
165  // we need vectors of indices and values for the Allgatherv, and a vector
166  // of pairs for the sort/merge
167  std::vector<INDEX> indices;
168  std::vector<SCALAR> values;
169  std::vector<std::pair<INDEX, SCALAR>> v;
170 
171  if (receiving)
172  {
173  s.resize(n*len);
174 
175  indices.resize(len);
176  values.resize(len);
177  v.resize(len);
178  }
179 
180 
181  // allreduce column-by-column
182  for (INDEX j=0; j<n; j++)
183  {
184  internal::get::col<INDEX, SCALAR>(j, x, a);
185 
186  // get the displacements
187  INDEX count_local = a.get_nnz();
188  mpi::gather(mpi::REDUCE_TO_ALL, &count_local, 1, counts.data_ptr(), 1, comm);
189 
190  unsigned int count = counts.sum();
191 
192  if (count == 0)
193  continue;
194  else if (receiving && indices.capacity() < count)
195  {
196  indices.resize(count);
197  values.resize(count);
198  v.resize(count);
199  }
200 
201  for (int i=1; i<displs.get_len(); i++)
202  displs[i] = displs[i-1] + counts[i-1];
203 
204  // get all the indices/values
205  mpi::gatherv(root, a.index_ptr(), a.get_nnz(), indices.data(), counts.data_ptr(), displs.data_ptr(), comm);
206  mpi::gatherv(root, a.data_ptr(), a.get_nnz(), values.data(), counts.data_ptr(), displs.data_ptr(), comm);
207 
208  // add all the vectors
209  if (receiving)
210  {
211  for (unsigned int i=0; i<count; i++)
212  v[i] = std::make_pair(indices[i], values[i]);
213 
214  std::sort(v.begin(), v.begin()+count);
215 
216  INDEX nnz = 0;
217  indices[0] = v[0].first;
218  values[0] = v[0].second;
219  for (unsigned int i=1; i<count; i++)
220  {
221  if (v[i].first == indices[nnz])
222  values[nnz] += v[i].second;
223  else
224  {
225  nnz++;
226 
227  indices[nnz] = v[i].first;
228  values[nnz] = v[i].second;
229  }
230  }
231 
232  // put summed column into the return
233  a.set(nnz+1, indices.data(), values.data());
234  s.insert(j, a);
235  }
236  }
237 
238  return s;
239  }
240  }
241 }
